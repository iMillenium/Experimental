-	script	::Airship_Staff_izlude	-1,{
	mes "[Assistente do Aeroplano]";
	mes "Bem-vindo!!";
	mes "Você gostaria de embarcar no";
	mes "Aeroplano, ele vai para Juno e";
	mes "Rachel.";
	next;
	if (select("Embarcar.:Cancelar.") == 1) {
		mes "[Assistente do Aeroplano]";
		mes "A passagem custa 1.200 zeny.";
		mes "Deseja comprar uma?";
		next;
		if (select("Sim:Não") == 1) {
			if (countitem(7311) > 0) {
				delitem 7311,1;
				warp "airplane_01",244,58;
				end;
			}
			if (Zeny >= 1200) {
				Zeny -= 1200;
				warp "airplane_01",244,58;
				end;
			}
			mes "[Assistente do Aeroplano]";
			mes "Me desculpe, mas você não possui";
			mes "1.200 zeny!";
			mes "Você não pode embarcar.";
			close;
		}
	}
	mes "[Assistente do Aeroplano]";
	mes "Muito obrigada!";
	mes "Volte sempre.";
	close;
}
